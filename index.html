import pygame import random # Inicialização pygame.init() # Configurações da Tela LARGURA = 400 ALTURA = 600 tela = pygame.display.set_mode((LARGURA, ALTURA)) pygame.display.set_caption("Jogo de Desviar - Corrida") # Cores BRANCO = (255, 255, 255) PRETO = (0, 0, 0) VERMELHO = (200, 0, 0) AZUL = (0, 0, 255) # Configurações do Jogador carro_largura = 50 carro_altura = 80 carro_x = (LARGURA * 0.45) carro_y = (ALTURA * 0.8) velocidade_jogador = 5 # Configurações do Obstáculo obs_largura = 50 obs_altura = 50 obs_x = random.randrange(0, LARGURA - obs_largura) obs_y = -600 obs_velocidade = 7 relogio = pygame.time.Clock() pontos = 0 rodando = True def mostrar_pontos(contagem): fonte = pygame.font.SysFont(None, 35) texto = fonte.render(f"Pontos: {contagem}", True, PRETO) tela.blit(texto, (10, 10)) # Loop Principal while rodando: for evento in pygame.event.get(): if evento.type == pygame.QUIT: rodando = False # Movimentação teclas = pygame.key.get_pressed() if teclas[pygame.K_LEFT] and carro_x > 0: carro_x -= velocidade_jogador if teclas[pygame.K_RIGHT] and carro_x < LARGURA - carro_largura: carro_x += velocidade_jogador # Lógica do Obstáculo obs_y += obs_velocidade # Renderização tela.fill(BRANCO) # Desenha Jogador e Obstáculo pygame.draw.rect(tela, AZUL, [carro_x, carro_y, carro_largura, carro_altura]) pygame.draw.rect(tela, VERMELHO, [obs_x, obs_y, obs_largura, obs_altura]) mostrar_pontos(pontos) # Verifica se o obstáculo saiu da tela if obs_y > ALTURA: obs_y = 0 - obs_altura obs_x = random.randrange(0, LARGURA - obs_largura) pontos += 1 obs_velocidade += 0.2 # Aumenta a dificuldade # Detecção de Colisão if carro_y < obs_y + obs_altura: if (carro_x > obs_x and carro_x < obs_x + obs_largura) or \ (carro_x + carro_largura > obs_x and carro_x + carro_largura < obs_x + obs_largura): print(f"Game Over! Sua pontuação: {pontos}") rodando = False pygame.display.update() relogio.tick(60) pygame.quit()
