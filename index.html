import pygame import sys # Inicializa o Pygame pygame.init() # Configurações da tela WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Jogo de Carro Mundo Aberto") # Cores WHITE = (255, 255, 255) GREEN = (0, 200, 0) # Carro (um retângulo simples) car_width, car_height = 50, 30 car_x, car_y = WIDTH // 2, HEIGHT // 2 car_speed = 5 # Loop principal clock = pygame.time.Clock() while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() # Teclas pressionadas keys = pygame.key.get_pressed() if keys[pygame.K_UP]: car_y -= car_speed if keys[pygame.K_DOWN]: car_y += car_speed if keys[pygame.K_LEFT]: car_x -= car_speed if keys[pygame.K_RIGHT]: car_x += car_speed # Desenha fundo e carro screen.fill(GREEN) # fundo verde como "grama" pygame.draw.rect(screen, WHITE, (car_x, car_y, car_width, car_height)) pygame.display.flip() clock.tick(60)
