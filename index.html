import * as THREE from 'three'; // --- CENA E RENDER --- const scene = new THREE.Scene(); scene.background = new THREE.Color(0x020205); scene.fog = new THREE.FogExp2(0x020205, 0.015); // Slightly softer fog const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000); const renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); renderer.toneMapping = THREE.ReinhardToneMapping; renderer.shadowMap.enabled = true; document.body.appendChild(renderer.domElement); // --- ILUMINAÇÃO --- scene.add(new THREE.AmbientLight(0x404040, 0.5)); const moonlight = new THREE.DirectionalLight(0xffffff, 0.5); moonlight.position.set(10, 20, 10); scene.add(moonlight); // --- CONSTRUÇÃO DO CARRO --- const carGroup = new THREE.Group(); const bodyMat = new THREE.MeshStandardMaterial({ color: 0x111111, metalness: 0.9, roughness: 0.1 }); const carBody = new THREE.Mesh(new THREE.BoxGeometry(1.2, 0.4, 2.6), bodyMat); carGroup.add(carBody); // Teto const glassMat = new THREE.MeshPhysicalMaterial({ color: 0x000000, metalness: 1, roughness: 0, transmission: 0.5 }); const cabin = new THREE.Mesh(new THREE.BoxGeometry(1, 0.4, 1.2), glassMat); cabin.position.set(0, 0.4, -0.2); carGroup.add(cabin); // Faróis Dianteiros const headlightGeom = new THREE.BoxGeometry(0.3, 0.1, 0.1); const headlightMat = new THREE.MeshBasicMaterial({ color: 0xffffff }); const hLeft = new THREE.Mesh(headlightGeom, headlightMat); hLeft.position.set(-0.4, 0.1, 1.3); const hRight = hLeft.clone(); hRight.position.x = 0.4; carGroup.add(hLeft, hRight); // Lanternas Traseiras (Brake Lights) const brakeLightGeom = new THREE.BoxGeometry(0.3, 0.1, 0.1); const brakeLightMat = new THREE.MeshBasicMaterial({ color: 0x330000 }); // Dim red by default const bLeft = new THREE.Mesh(brakeLightGeom, brakeLightMat); bLeft.position.set(-0.4, 0.1, -1.3); const bRight = bLeft.clone(); bRight.position.x = 0.4; carGroup.add(bLeft, bRight); // Rodas const wheels = []; const wheelGeom = new THREE.CylinderGeometry(0.35, 0.35, 0.2, 24); const wheelMat = new THREE.MeshStandardMaterial({ color: 0x050505, roughness: 0.8 }); [[-0.7, -0.1, 0.8], [0.7, -0.1, 0.8], [-0.7, -0.1, -0.8], [0.7, -0.1, -0.8]].forEach(pos => { const w = new THREE.Mesh(wheelGeom, wheelMat); w.rotation.z = Math.PI / 2; w.position.set(...pos); carGroup.add(w); wheels.push(w); }); scene.add(carGroup); // --- ESTRADA INFINITA --- const grid = new THREE.GridHelper(2000, 100, 0x00ffcc, 0x222222); scene.add(grid); // --- FÍSICA E CONTROLES --- let speed = 0; let distance = 0; const keys = {}; window.addEventListener('keydown', e => keys[e.code] = true); window.addEventListener('keyup', e => keys[e.code] = false); function update() { const isBraking = keys['Space'] || keys['KeyS'] || keys['ArrowDown']; // Logic: If braking and moving forward, light up bright red if (isBraking && speed > 0.01) { brakeLightMat.color.setHex(0xff0000); } else { brakeLightMat.color.setHex(0x330000); } // Aceleração if (keys['KeyW'] || keys['ArrowUp']) speed += 0.008; if (keys['KeyS'] || keys['ArrowDown']) speed -= 0.012; if (keys['Space']) speed *= 0.92; // Handbrake speed *= 0.985; // Air Resistance // Steering if (Math.abs(speed) > 0.01) { // Steering is more sensitive at medium speeds, harder at high speeds const turnAbility = Math.min(Math.abs(speed) * 2, 0.04); const direction = speed > 0 ? 1 : -1; if (keys['KeyA'] || keys['ArrowLeft']) carGroup.rotation.y += turnAbility * direction; if (keys['KeyD'] || keys['ArrowRight']) carGroup.rotation.y -= turnAbility * direction; } carGroup.translateZ(speed); // INFINITE WORLD LOGIC // Move the grid with the car so the player never reaches the edge grid.position.x = carGroup.position.x; grid.position.z = carGroup.position.z; // Wheel Spin wheels.forEach(w => w.rotation.x += speed * 2.5); // Camera const camDist = -7 - (speed * 5); const camHeight = 2.5 + (speed * 2); const relativeCameraOffset = new THREE.Vector3(0, camHeight, camDist); const cameraOffset = relativeCameraOffset.applyMatrix4(carGroup.matrixWorld); camera.position.lerp(cameraOffset, 0.1); camera.lookAt(carGroup.position.clone().add(new THREE.Vector3(0, 1, 0))); // UI distance += Math.abs(speed); document.getElementById('kmh').innerText = Math.floor(Math.abs(speed) * 450); document.getElementById('dist').innerText = Math.floor(distance); } function animate() { requestAnimationFrame(animate); update(); renderer.render(scene, camera); } window.addEventListener('resize', () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); }); animate();
